turtles-own [ know-gossip]
globals [ in-the-know  number-to-learn-news]

to go
  count-turtles-with-news
  if in-the-know >= Max-Number-Of-Students [ stop ]
  spread-gossip
  tick
end

to count-turtles-with-news
  set in-the-know count turtles
  print in-the-know
end

to spread-gossip
  if Model = 1 [                           ;; linear model
    set Gossip-Spread-Rate 1
    create-turtles Gossip-Spread-Rate
    let x (count turtles)
    print x
    ask turtle (x - 1) [
      set know-gossip true
      set shape "wolf 3"
      set color red
      set size 1.5
      setxy random-xcor random-ycor
    ]
    ;;ask turtles [ setxy random xcor random ycor ]
    count-turtles-with-news
  ]
   if Model = 2 [               ;; make this one exponential
    if Gossip-Spread-Rate = 1 [ set Gossip-Spread-Rate 1.1 ]
    set number-to-learn-news ( in-the-know * (Gossip-Spread-Rate ^ ticks)  )
    set number-to-learn-news (number-to-learn-news - in-the-know )
    repeat number-to-learn-news [
      create-turtles 1
      ask turtle ( count turtles ) [set know-gossip true ]
      ask turtle ( count turtles ) [set shape "wolf3" ]
      ask turtle ( count turtles ) [set color red ]
      ask turtle ( count turtles ) [setxy random-xcor random-ycor ]
      count-turtles-with-news
        ]
      ]
  if Model = 3 [
    set Gossip-Spread-Rate 2
    set number-to-learn-news (ticks ^ Gossip-Spread-Rate )
    set number-to-learn-news (number-to-learn-news - in-the-know )
   repeat number-to-learn-news [
      create-turtles 1
      ask turtle ( count turtles ) [set know-gossip true ]
      ask turtle ( count turtles ) [set shape "wolf3" ]
      ask turtle ( count turtles ) [set color red ]
      ask turtle ( count turtles ) [setxy random-xcor random-ycor ]
      count-turtles-with-news
    ]
  ]

end

to reset
  clear-all
  setup-patches
  setup-turtles
  reset-ticks
end

to setup-turtles
  clear-all
  reset-ticks
  setup-patches
  create-turtles 1
  ask turtles [ setxy ( max-pxcor - min-pxcor ) / 2 ( max-pycor - min-pycor) / 2]
  ask turtles [ set shape "person student" ]
  ask turtles [ set color blue ]
  ask turtles [ set size 1.5]
  ;;ask turtle 1 [ set color red ]
  count-turtles-with-news
end

to setup-patches
  ask patches [ set pcolor grey ]
end
